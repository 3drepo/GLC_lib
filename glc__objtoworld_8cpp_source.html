69 GLC_World* GLC_ObjToWorld::CreateWorldFromObj(QFile &file)
70 {
71  m_ListOfAttachedFileName.clear();

74  // Test if the file exist and can be opened
76  if (!file.open(QIODevice::ReadOnly))
77  {
78  QString message(QString("GLC_ObjToWorld::CreateWorldFromObj File ") + m_FileName + QString(" doesn't exist"));
79  //qDebug() << message;

81  throw(fileFormatException);

83 
85  // Init member
87  m_pWorld= new GLC_World;
88 
89  // Create Working variables
90  int currentQuantumValue= 0;
91  int previousQuantumValue= 0;
92  int numberOfLine= 0;
93 
94  // Create the input file stream
95  QTextStream objStream(&file);
96 
97  // QString buffer
98  QString lineBuff;
99 
100  QString mtlLibLine;
101 
103  // Searching mtllib attribute
105  while (!objStream.atEnd() && !lineBuff.contains("mtllib"))
106  {
107  ++numberOfLine;
108  lineBuff= objStream.readLine();
109  if (lineBuff.contains("mtllib")) mtlLibLine= lineBuff;
110  }
111 
113  // Count the number of lines of the OBJ file
115  while (!objStream.atEnd())
116  {
117  ++numberOfLine;
118  objStream.readLine();
119  }
120 
122  // Reset the stream
124  objStream.resetStatus();
125  objStream.seek(0);
126 
128  // if mtl file found, load it
130  QString mtlLibFileName(getMtlLibFileName(mtlLibLine));
131  if (!mtlLibFileName.isEmpty())
132  {
133  m_pMtlLoader= new GLC_ObjMtlLoader(mtlLibFileName);
134  if (!m_pMtlLoader->loadMaterials())

136  delete m_pMtlLoader;

174  }

177  }

231 void GLC_ObjToWorld::scanLigne(QString &line)
232 {
233  line= line.trimmed();
234  // Search Vertexs vectors
235  if (line.startsWith("v ")|| line.startsWith(QString("v") + QString(QChar(9))))
236  {
237  line.remove(0,2); // Remove first 2 char
238  m_Positions.append(extract3dVect(line));
239  m_FaceType = notSet;
240  }
241 
242  // Search texture coordinate vectors
243  else if (line.startsWith("vt ")|| line.startsWith(QString("vt") + QString(QChar(9))))
244  {
245  line.remove(0,3); // Remove first 3 char
246  m_Texels.append(extract2dVect(line));
247  m_FaceType = notSet;
248  }
249 
250  // Search normals vectors
251  else if (line.startsWith("vn ") || line.startsWith(QString("vn") + QString(QChar(9))))
252  {
253  line.remove(0,3); // Remove first 3 char
254  m_Normals.append(extract3dVect(line));
255  m_FaceType = notSet;
256  }
257 
258  // Search faces to update index
259  else if (line.startsWith("f ") || line.startsWith(QString("f") + QString(QChar(9))))
260  {
261  // If there is no group or object in the OBJ file
262  if (NULL == m_pCurrentObjMesh)
263  {
264  changeGroup("GLC_Default");
265  //qDebug() << "Default group " << line;
266  }
267  line.remove(0,2); // Remove first 2 char
268  extractFaceIndex(line);
269  }
270 
271  // Search Material
272  else if (line.startsWith("usemtl ") || line.startsWith(QString("usemtl") + QString(QChar(9))))
273  {
274  line.remove(0,7); // Remove first 7 char
275  setCurrentMaterial(line);
276  m_FaceType = notSet;
277  }
278 
279  // Search Group
280  else if (line.startsWith("g ") || line.startsWith("o ") || line.startsWith(QString("g") + QString(QChar(9)))
281  || line.startsWith(QString("o") + QString(QChar(9))))
282  {
283  m_FaceType = notSet;
284  line.remove(0,2); // Remove first 2 char
285  changeGroup(line);
286  }

401 // Extract a face from a string
402 void GLC_ObjToWorld::extractFaceIndex(QString &line)
403 {
404  QString buff;
405 
406  int coordinateIndex;
407  int normalIndex;
408  int textureCoordinateIndex;
409 
410  QList<GLuint> currentFaceIndex;
412  // Parse the line containing face index
414  QTextStream streamFace(&line);
415  while ((!streamFace.atEnd()))
416  {
417  streamFace >> buff;
418  extractVertexIndex(buff, coordinateIndex, normalIndex, textureCoordinateIndex);
419 
420  ObjVertice currentVertice(coordinateIndex, normalIndex, textureCoordinateIndex);
421  if (m_pCurrentObjMesh->m_ObjVerticeIndexMap.contains(currentVertice))
422  {
423  currentFaceIndex.append(m_pCurrentObjMesh->m_ObjVerticeIndexMap.value(currentVertice));
424  }
425  else
426  {
427  // Add Vertex to the mesh bulk data
428  m_pCurrentObjMesh->m_Positions.append(m_Positions.value(coordinateIndex * 3));
429  m_pCurrentObjMesh->m_Positions.append(m_Positions.value(coordinateIndex * 3 + 1));
430  m_pCurrentObjMesh->m_Positions.append(m_Positions.value(coordinateIndex * 3 + 2));
431  if (-1 != normalIndex)
432  {
433  // Add Normal to the mesh bulk data
434  m_pCurrentObjMesh->m_Normals.append(m_Normals.value(normalIndex * 3));
435  m_pCurrentObjMesh->m_Normals.append(m_Normals.value(normalIndex * 3 + 1));
436  m_pCurrentObjMesh->m_Normals.append(m_Normals.value(normalIndex * 3 + 2));
437  }
438  else
439  {
440  // Add Null Normal to the mesh bulk data
441  m_pCurrentObjMesh->m_Normals.append(0.0f);
442  m_pCurrentObjMesh->m_Normals.append(0.0f);
443  m_pCurrentObjMesh->m_Normals.append(0.0f);
444  }
445  if (-1 != textureCoordinateIndex)
446  {
447  // Add texture coordinate to the mesh bulk data
448  m_pCurrentObjMesh->m_Texels.append(m_Texels.value(textureCoordinateIndex * 2));
449  m_pCurrentObjMesh->m_Texels.append(m_Texels.value(textureCoordinateIndex * 2 + 1));
450  }
451  else if (!m_pCurrentObjMesh->m_Texels.isEmpty())
452  {
453  // Add epmty texture coordinate
454  m_pCurrentObjMesh->m_Texels.append(0.0f);
455  m_pCurrentObjMesh->m_Texels.append(0.0f);
456  }
457  // Add the index to current face index
458  currentFaceIndex.append(m_pCurrentObjMesh->m_NextFreeIndex);
459  // Add ObjVertice to ObjVertice Map
460  m_pCurrentObjMesh->m_ObjVerticeIndexMap.insert(currentVertice, m_pCurrentObjMesh->m_NextFreeIndex);
461  // Increment next free index
462  ++(m_pCurrentObjMesh->m_NextFreeIndex);
463  }
464 
465  }
467  // Check the number of face's vertex
469  const int size= currentFaceIndex.size();
470  if (size < 3)
471  {
472  QStringList stringList(m_FileName);
473  stringList.append("GLC_ObjToWorld::extractFaceIndex Face with less than 3 vertex found");
474  GLC_ErrorLog::addError(stringList);
475  return;
476  }
478  // Add the face to the current mesh
480  if ((m_FaceType == coordinateAndNormal) || (m_FaceType == coordinateAndTextureAndNormal))
481  {
482  if (size > 3)
483  {
484  glc::triangulatePolygon(&currentFaceIndex, m_pCurrentObjMesh->m_Positions);
485  }
486  m_pCurrentObjMesh->m_Index.append(currentFaceIndex);
487  }
488  else if (m_FaceType != notSet)
489  {
490  if (size > 3)
491  {
492  glc::triangulatePolygon(&currentFaceIndex, m_pCurrentObjMesh->m_Positions);
493  }
494  // Comput the face normal
495  if (currentFaceIndex.size() < 3) return;
496  GLC_Vector3df normal= computeNormal(currentFaceIndex.at(0), currentFaceIndex.at(1), currentFaceIndex.at(2));
497 
498  // Add Face normal to bulk data
499  QSet<GLuint> indexSet= currentFaceIndex.toSet();
500  QSet<GLuint>::iterator iIndexSet= indexSet.begin();
501  while (indexSet.constEnd() != iIndexSet)
502  {
503  m_pCurrentObjMesh->m_Normals[*iIndexSet * 3]= normal.x();
504  m_pCurrentObjMesh->m_Normals[*iIndexSet * 3 + 1]= normal.y();
505  m_pCurrentObjMesh->m_Normals[*iIndexSet * 3 + 2]= normal.z();
506 
507  ++iIndexSet;
508  }
509 
510  m_pCurrentObjMesh->m_Index.append(currentFaceIndex);
511 
512  }
513  else
514  {
515  QString message= "GLC_ObjToWorld::extractFaceIndex " + m_FileName + " unknow face type";
516  message.append("\nAt line : ");
517  message.append(QString::number(m_CurrentLineNumber));
518  GLC_FileFormatException fileFormatException(message, m_FileName, GLC_FileFormatException::FileNotSupported);
519  clear();
520  throw(fileFormatException);
521  }

525 {

566  }
567  }
568  }
569  // Create this material offsetSize

688  clear();
689  throw(fileFormatException);
690  }
691  }
692  else if (m_FaceType == coordinate)

769 // compute face normal
770 GLC_Vector3df GLC_ObjToWorld::computeNormal(GLuint index1, GLuint index2, GLuint index3)
771 {
772  double xn, yn, zn;
773 
774  // Vertex 1
775  xn= m_pCurrentObjMesh->m_Positions.at(index1 * 3);
776  yn= m_pCurrentObjMesh->m_Positions.at(index1 * 3 + 1);
777  zn= m_pCurrentObjMesh->m_Positions.at(index1 * 3 + 2);
778  const GLC_Vector3d vect1(xn, yn, zn);
779 
780  // Vertex 2
781  xn= m_pCurrentObjMesh->m_Positions.at(index2 * 3);
782  yn= m_pCurrentObjMesh->m_Positions.at(index2 * 3 + 1);
783  zn= m_pCurrentObjMesh->m_Positions.at(index2 * 3 + 2);
784  const GLC_Vector3d vect2(xn, yn, zn);
785 
786  // Vertex 3
787  xn= m_pCurrentObjMesh->m_Positions.at(index3 * 3);
788  yn= m_pCurrentObjMesh->m_Positions.at(index3 * 3 + 1);
789  zn= m_pCurrentObjMesh->m_Positions.at(index3 * 3 + 2);
790  const GLC_Vector3d vect3(xn, yn, zn);
791 
792  const GLC_Vector3d edge1(vect3 - vect2);
793  const GLC_Vector3d edge2(vect1 - vect2);
794 
795  GLC_Vector3d normal(edge1 ^ edge2);
796  normal.normalize();
797 
798  return normal.toVector3df();
799 }

831 // Add the current Obj mesh to the world
832 void GLC_ObjToWorld::addCurrentObjMeshToWorld()
833 {
834  if (NULL != m_pCurrentObjMesh)
835  {
836  if (!m_pCurrentObjMesh->m_Positions.isEmpty())
837  {
838  m_pCurrentObjMesh->m_pMesh->addVertice(m_pCurrentObjMesh->m_Positions.toVector());
839  m_pCurrentObjMesh->m_Positions.clear();
840  m_pCurrentObjMesh->m_pMesh->addNormals(m_pCurrentObjMesh->m_Normals.toVector());
841  m_pCurrentObjMesh->m_Normals.clear();
842  if (!m_pCurrentObjMesh->m_Texels.isEmpty())
843  {
844  m_pCurrentObjMesh->m_pMesh->addTexels(m_pCurrentObjMesh->m_Texels.toVector());
845  m_pCurrentObjMesh->m_Texels.clear();
846  }
847  QHash<QString, MatOffsetSize*>::iterator iMat= m_pCurrentObjMesh->m_Materials.begin();
848  while (m_pCurrentObjMesh->m_Materials.constEnd() != iMat)
849  {
850  GLC_Material* pCurrentMaterial= NULL;
851  if ((NULL != m_pMtlLoader) && (m_pMtlLoader->contains(iMat.key())))
852  {
853  pCurrentMaterial= m_pMtlLoader->material(iMat.key());
854  }
855  // Create the list of triangles to add to the mesh
856  const int offset= iMat.value()->m_Offset;
857  int size= iMat.value()->m_size;
858  if (0 == size)
859  {
860  size= m_pCurrentObjMesh->m_Index.size() - offset;
861  }
862  //qDebug() << "Offset : " << offset << " size : " << size;
863  QList<GLuint> triangles;
864  for (int i= offset; i < (offset + size); ++i)
865  {
866  triangles.append(m_pCurrentObjMesh->m_Index.at(i));
867  }
868  // Add the list of triangle to the mesh
869  if (!triangles.isEmpty())
870  {
871  m_pCurrentObjMesh->m_pMesh->addTriangles(pCurrentMaterial, triangles);
872  }
873 
874  ++iMat;
875  }
876  if (m_pCurrentObjMesh->m_pMesh->faceCount(0) > 0)
877  {
878  m_pCurrentObjMesh->m_pMesh->finish();
879  GLC_3DRep* pRep= new GLC_3DRep(m_pCurrentObjMesh->m_pMesh);
880  m_pWorld->rootOccurence()->addChild((new GLC_StructInstance(pRep)));
881  }
882  else
883  {
884  delete m_pCurrentObjMesh->m_pMesh;
885  }
886 
887  }
888  else
889  {
890  delete m_pCurrentObjMesh->m_pMesh;
891  }
892 
893  delete m_pCurrentObjMesh;
894  m_pCurrentObjMesh= NULL;
895  }
896 }
