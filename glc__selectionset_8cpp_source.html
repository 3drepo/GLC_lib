70 {
71  Q_ASSERT(m_pWorldHandle->containsOccurence(occurenceId));
72  if (!m_OccurenceHash.contains(occurenceId))
73  {
74  GLC_StructOccurence* pOccurence= m_pWorldHandle->getOccurence(occurenceId);
