411 GLC_Matrix4x4 GLC_Viewport::compositionMatrix() const
412 {
413  // Get the viewport projection matrix
414  GLC_Matrix4x4 projectionMatrix= m_ProjectionMatrix;
415  // Get the camera modelView matrix
416  GLC_Matrix4x4 modelViewMatrix= m_pViewCam->modelViewMatrix();
417  // Composition matrix
418  GLC_Matrix4x4 compMatrix= projectionMatrix * modelViewMatrix;
419 
420  return compMatrix;
421 }

423 bool GLC_Viewport::updateFrustum(GLC_Matrix4x4* pMat)
424 {
425  if (NULL == pMat)
