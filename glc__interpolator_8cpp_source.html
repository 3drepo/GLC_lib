115 bool GLC_Interpolator::CalcInterpolLineaireMat(void)
116 {
117 
118  const GLC_Vector3d VectTrans= (m_EndPoint - m_StartPoint) * (1.0 / m_StepCount);
119  if(VectTrans.isNull())
120  {
121  m_InterpolMat.setToIdentity();
122  return false;
123  }
124  else
125  {
126  m_InterpolMat.setMatTranslate(VectTrans);
127  return true;
128  }
129 }
