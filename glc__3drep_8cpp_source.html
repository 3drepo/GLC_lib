431 {

465 QDataStream &operator>>(QDataStream & stream, GLC_3DRep & rep)
466 {
467  Q_ASSERT(rep.isEmpty());
468 
469  quint32 chunckId;
470  stream >> chunckId;
471  Q_ASSERT(chunckId == GLC_3DRep::m_ChunkId);
472 
473  // The rep name
474  QString name;
475  stream >> name;
476  rep.setName(name);
477 
478  // Retrieve the list of rep materials
479  QList<GLC_Material> materialsList;
480  stream >> materialsList;
481  MaterialHash materialHash;
482  // Update mesh materials hash table
483  QHash<GLC_uint, GLC_uint> materialIdMap;
484  const int materialsCount= materialsList.size();
485  for (int i= 0; i < materialsCount; ++i)
486  {
487  GLC_Material* pMaterial= new GLC_Material(materialsList.at(i));
488  pMaterial->setId(glc::GLC_GenID());
489  materialIdMap.insert(materialsList.at(i).id(), pMaterial->id());
490  materialHash.insert(pMaterial->id(), pMaterial);
491  }
492 
493  int meshNumber;
494  stream >> meshNumber;
495  for (int i= 0; i < meshNumber; ++i)
496  {
497  GLC_Mesh* pMesh= new GLC_Mesh();
498  pMesh->loadFromDataStream(stream, materialHash, materialIdMap);
499 
500  rep.addGeom(pMesh);
501  }
502 
503  return stream;
